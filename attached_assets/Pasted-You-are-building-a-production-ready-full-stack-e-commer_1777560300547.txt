export function getB2cDiscount(ordersCount: number) {
  if (ordersCount === 0) return { percent: 15, label: "First Order — 15% Off", color: "amber" }
  if (ordersCount === 1) return { percent: 10, label: "Loyalty Discount — 10% Off", color: "teal" }
  return { percent: 5, label: "Regular Discount — 5% Off", color: "slate" }
}

export function calculateB2cTotal(items: CartItem[], ordersCount: number) {
  const subtotal = items.reduce((sum, i) => sum + i.unitPrice * i.quantity, 0)
  const { percent } = getB2cDiscount(ordersCount)
  const discountAmount = parseFloat(((subtotal * percent) / 100).toFixed(2))
  const afterDiscount = subtotal - discountAmount
  const gstAmount = parseFloat(((afterDiscount * 5) / 100).toFixed(2))
  const shippingCharge = afterDiscount >= 999 ? 0 : 60
  const total = afterDiscount + gstAmount + shippingCharge
  return { subtotal, discountAmount, discountPercent: percent, gstAmount, shippingCharge, total }
}

export function calculateB2bTotal(items: CartItem[]) {
  const subtotal = items.reduce((sum, i) => sum + i.unitPrice * i.quantity, 0)
  if (subtotal < 3000) throw new Error("Minimum B2B order value is ₹3,000")
  const gstAmount = parseFloat(((subtotal * 5) / 100).toFixed(2))
  const total = subtotal + gstAmount
  return { subtotal, gstAmount, total }
}

export function validateB2bMoq(items: CartItem[]) {
  const violations = items.filter(i => i.quantity < i.product.moq)
  if (violations.length > 0) {
    throw new Error(`MOQ not met for: ${violations.map(v => `${v.product.name} (min ${v.product.moq})`).join(', ')}`)
  }
}


`src/store/cartStore.ts`
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

interface CartItem {
  product: Product
  quantity: number
  unitPrice: number
}

interface CartStore {
  items: CartItem[]
  orderType: 'b2c' | 'b2b'
  isOpen: boolean
  addItem: (product: Product, unitPrice: number, qty?: number) => void
  removeItem: (productId: string) => void
  updateQty: (productId: string, qty: number) => void
  clearCart: () => void
  toggleCart: () => void
  setOrderType: (type: 'b2c' | 'b2b') => void
}

export const useCartStore = create<CartStore>()(
  persist(
    (set, get) => ({
      items: [],
      orderType: 'b2c',
      isOpen: false,
      addItem: (product, unitPrice, qty = 1) => {
        const items = get().items
        const existing = items.find(i => i.product.id === product.id)
        if (existing) {
          set({ items: items.map(i => i.product.id === product.id ? { ...i, quantity: i.quantity + qty } : i) })
        } else {
          set({ items: [...items, { product, quantity: qty, unitPrice }] })
        }
      },
      removeItem: (productId) => set({ items: get().items.filter(i => i.product.id !== productId) }),
      updateQty: (productId, qty) => {
        if (qty <= 0) { get().removeItem(productId); return }
        set({ items: get().items.map(i => i.product.id === productId ? { ...i, quantity: qty } : i) })
      },
      clearCart: () => set({ items: [] }),
      toggleCart: () => set({ isOpen: !get().isOpen }),
      setOrderType: (type) => set({ orderType: type }),
    }),
    { name: 'snackveda-cart' }
  )
)

  const response = NextResponse.next()
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    { cookies: { get: (name) => request.cookies.get(name)?.value, set: (name, value, options) => response.cookies.set(name, value, options), remove: (name, options) => response.cookies.delete(name) } }
  )
  const { data: { session } } = await supabase.auth.getSession()

  // Protect account routes
  if (request.nextUrl.pathname.startsWith('/account') && !session) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  // Protect admin routes - check role in DB
  if (request.nextUrl.pathname.startsWith('/admin')) {
    if (!session) return NextResponse.redirect(new URL('/login', request.url))
    // Role check happens in each admin page/API
  }

  return response
}

export const config = { matcher: ['/account/:path*', '/admin/:path*', '/checkout'] }
